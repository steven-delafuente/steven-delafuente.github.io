#Create polygon from state shappefile.  

def create_polygon(data_loc):
    
    poly_shp = pd.read_csv(data_loc)
    #reverse dataframe colum order.
    subset_list = poly_shp[['Lon', 'Lat']]
    # Create tuple of properly ofrmated x & y (Lon, Lat) values.
    shp_List = [tuple(x) for x in subset_list.values]
    #create shapely polygon.
    poly = Polygon(shp_List)
    return(poly)

# Create data mask using spatial polygon. Used for regional analysis. 

def mask_output(data,lat_list, lon_list, polygon):
    Mask = np.zeros(data.shape[1:])
    
    for lo_i in range(len(lon_list)):
        # iterate through latitude array
        for la_i in range(len(lat_list)):
            # set current lon/lat
            lo = lon_list[lo_i]  # current longitude coord in iteration
            la = lat_list[la_i]  # current latitude coord in iteration
            # create current point
            currentPoint = Point(lo,la)
            #ca_point.append(currentPoint)
            # check if point inside: returns true or false
            check = currentPoint.within(polygon)
            Mask[la_i,lo_i] = int(check)
        
    return(Mask)

#Class to calculate common statistics by grid cell. 
class stat_calc():
    # Maximum temp anomaly per year
    def max_temp(data,data_mask, time_delta):
        temp = np.zeros(time_delta)
        for t in range(time_delta):
            temp[t] = np.nanmax(data[t]*data_mask) 
        return(temp)
    # Minimun tempature anomaly per year 
    def min_temp(data,data_mask, time_delta):
        temp = np.zeros(time_delta)
        for t in range(time_delta):
            temp[t] = np.nanmin(data[t]*data_mask) 
        return(temp)
    
    def mean_temp(data,data_mask, time_delta):
        temp = np.zeros(time_delta)
        for t in range(time_delta):
            temp[t] = np.nanmean(data[t]*data_mask) 
        return(temp)
